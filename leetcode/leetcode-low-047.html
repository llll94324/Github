帕斯卡三角形

給定一個非負整數 numRows，生成楊輝三角的前 numRows 行。

輸入: 5
輸出:
[
     [1],
    [1,1],
   [1,2,1],
  [1,3,3,1],
 [1,4,6,4,1]
]
      
  /**
   * @param {number} numRows
   * @return {number[][]}
   */
  var generate = function(numRows) {
     if (numRows == 0) return []
      if (numRows == 1) return [[1]]
      if (numRows == 2) return [[1], [1, 1]]
      let ans = [[1], [1, 1]]
      for (let len = 2; len < numRows; len++) {
          let row = [1]
          for (let i = 1; i < len; i++) {
              row.push(ans[len - 1][i - 1] + ans[len - 1][i])
          }
          row.push(1)
          ans.push(row)
      }
      return ans
  };
