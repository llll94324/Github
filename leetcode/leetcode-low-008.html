移動零

給定一個數組 nums，編寫一個函數將所有 0 移動到數組的末尾，同時保持非零元素的相對順序。

輸入: [0,1,0,3,12]
輸出: [1,3,12,0,0]
      
   /**
   * @param {number[]} nums
   * @return {void} Do not return anything, modify nums in-place instead.
   */
  var moveZeroes = function(nums) {
      let l = nums.length;
      for(let i = 0 ; i < l ;){
          if(nums[i] === 0){
              nums.splice(i,1);
              nums.push(0);
              l--
          }else{
              i++;
          }
      }
  };
