顏色分類

給定一個包含紅色、白色和藍色，一共 n 個元素的數組，原地對它們進行排序，使得相同顏色的元素相鄰，並按照紅色、白色、藍色順序排列。

此題中，我們使用整數 0、 1 和 2 分別表示紅色、白色和藍色。

輸入: [2,0,2,1,1,0]
輸出: [0,0,1,1,2,2]
      
  /**
   * @param {number[]} nums
   * @return {void} Do not return anything, modify nums in-place instead.
   */
  function swap(nums, idx1, idx2) {
      let temp = nums[idx1];
      nums[idx1] = nums[idx2];
      nums[idx2] = temp;
  }

  function sortColors(nums){
    var left = 0
    var right = nums.length - 1
    while(left < right && nums[left] == 0){
      left++
    }
    while(left < right && nums[right] == 2){
      right--
    }

    for(var i = left; i <= right;){
      if(nums[i] == 2){
        swap(nums,i,right)
        right--
      }else if(nums[i] == 0){
        if(left == i){
          left++
          i++
        }else{
          swap(nums,i,left)
          left++
        }
      }else{
        i++
      }
    }
  }
