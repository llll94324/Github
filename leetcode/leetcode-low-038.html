Shuffle an Array

打亂一個沒有重復元素的數組。

// 以數字集合 1, 2 和 3 初始化數組。
int[] nums = {1,2,3};
Solution solution = new Solution(nums);

// 打亂數組 [1,2,3] 並返回結果。任何 [1,2,3]的排列返回的概率應該相同。
solution.shuffle();

// 重設數組到它的初始狀態[1,2,3]。
solution.reset();

// 隨機返回數組[1,2,3]打亂後的結果。
solution.shuffle();

      
  /**
   * @param {number[]} nums
   */
  var Solution = function(nums) {
      this.resetVal  = [...nums];
      this.nums = [...nums];
  };

  /**
   * Resets the array to its original configuration and return it.
   * @return {number[]}
   */
  Solution.prototype.reset = function() {
      return this.resetVal
  };

  /**
   * Returns a random shuffling of the array.
   * @return {number[]}
   */
  Solution.prototype.shuffle = function() {
      this.resetVal = this.nums;
      let nums = this.nums.slice(),
          n = nums.length,i;
      while(n){
          i = Math.floor(Math.random()*n--);
          let t = nums[i];
          nums[i] = nums[n];
          nums[n] = t;
      }
      return nums
  };

   * var param_1 = obj.reset()
   * var param_2 = obj.shuffle()
