只出現一次的數字

給定一個非空整數數組，除了某個元素只出現一次以外，其余每個元素均出現兩次。找出那個只出現了一次的元素。

輸入: [2,2,1]
輸出: 1

      
  /**
   * @param {number[]} nums
   * @return {number}
   */
  var singleNumber = function(nums) {
      var arr = [...nums];
      var c = 0;
      for(var i = 0; i < nums.length;i++){
          if(nums.indexOf(nums[i]) != i){
              arr.splice(arr.indexOf(nums[i]),1);
              arr.splice(arr.indexOf(nums[i]),1);
          }
      }
      return arr[0];
  };
