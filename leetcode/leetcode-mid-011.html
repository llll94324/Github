二叉樹的鋸齒形層次遍歷

給定一個二叉樹，返回其節點值的鋸齒形層次遍歷。（即先從左往右，再從右往左進行下一層遍歷，以此類推，層與層之間交替進行）。

      
  /**
   * @param {TreeNode} root
   * @return {number[][]}
   */
  var zigzagLevelOrder = function (root) {
      sol = []
      travel(root, sol, 0)
      return sol
  };

  function travel(node, sol, height) {
    if(!node) return null
    if(height >= sol.length) sol.push([])
    row = sol[height]
    if(height % 2 == 0) row.push(node.val)
    else row.unshift(node.val) 
    
    travel(node.left,sol,height+1)
    travel(node.right,sol,height+1)
  }
