前 K 個高頻元素

給定一個非空的整數數組，返回其中出現頻率前 k 高的元素。

輸入: nums = [1,1,1,2,2,3], k = 2
輸出: [1,2]

      
  /**
   * @param {number[]} nums
   * @param {number} k
   * @return {number[]}
   */
  var topKFrequent = function(nums, k) {
    if(k === nums.length) return nums
    var map = new Map()
    var res = []
    var arr = []
    nums.forEach(val => map.set(val,(map.get(val) | 0) + 1))

    for(var [index,numbers] of map){
      if(arr[numbers] == undefined){
        arr[numbers] = index;
      }else{
        if(arr[numbers] instanceof Array){
          arr[numbers].push(index)
        }else{
          arr[numbers] = [arr[numbers],index]
        }
      }
    }

    var i = arr.length - 1

    while(k > 0){
      if(arr[i] != undefined){
        if(arr[i] instanceof Array){
          res.push(...arr[i])
          k -= arr[i].length
        }else{
          res.push(arr[i])
          k--
        }
      }
      i--
    }

    return res
  };
