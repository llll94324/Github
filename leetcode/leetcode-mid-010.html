中序遍歷二叉樹

給定一個二叉樹，返回它的中序 遍歷。

輸入: [1,null,2,3]
   1
    \
     2
    /
   3

輸出: [1,3,2]
      
  /**
   * Definition for a binary tree node.
   * function TreeNode(val) {
   *     this.val = val;
   *     this.left = this.right = null;
   * }
   */
  /**
   * @param {TreeNode} root
   * @return {number[]}
   */

  var inorderTraversal = function(node) {
      var r = []
    
      mid(node,r)
   
      return r
  };

  function mid(node,r){
  if (node) {
      mid(node.left,r);
      r.push(node.val)
      mid(node.right,r);
    }
  }
