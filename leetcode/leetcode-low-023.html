反轉鏈表

反轉一個單鏈表。

輸入: 1->2->3->4->5->NULL
輸出: 5->4->3->2->1->NULL
      
  /**
   * Definition for singly-linked list.
   * function ListNode(val) {
   *     this.val = val;
   *     this.next = null;
   * }
   */
  /**
   * @param {ListNode} head
   * @return {ListNode}
   */
  var reverseList = function(head) {
      let ans = null;
      let cur = head;
      while (cur != null) {
          let nextTmp = cur.next
          cur.next = ans
          ans = cur
          cur = nextTmp
      }
      return ans
  };
