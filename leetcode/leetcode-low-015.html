有效的字母異位詞

給定兩個字符串 s 和 t ，編寫一個函數來判斷 t 是否是 s 的字母異位詞。

輸入: s = "anagram", t = "nagaram"
輸出: true

      
  /**
   * @param {string} s
   * @param {string} t
   * @return {boolean}
   */
  var isAnagram = function (s, t) {
      var len = s.length;
      const map = {
      a: 0, b: 0, c: 0, d: 0, e: 0, f: 0, g: 0, h: 0, i: 0, j: 0, k: 0, l: 0, m: 0, n: 0, o: 0, p: 0, q: 0, r: 0, s: 0, t: 0, u: 0, v: 0, w: 0, x: 0, y: 0, z: 0
  };
      if (len !== t.length) {
          return false;
      }
      for (let i = 0; i < len; i++) {
          map[s[i]]++;
          map[t[i]]--;
      }
      for (let k in map) {
          if (map[k] !== 0) {
              return false;
          }
      }

      return true;
  };
