字符串轉換整數 (atoi)

請你來實現一個 atoi 函數，使其能將字符串轉換成整數。

輸入: "4193 with words"
輸出: 4193
解釋: 轉換截止於數字 '3' ，因為它的下一個字符不為數字。

      
  /**
   * @param {string} str
   * @return {number}
   */
  var myAtoi = function(str) {
      str = str.trim().match(/^(-?||\+?)\d+/)
      if(str == null){
         return 0;
      }
      str = str[0];
      
      var r = true;
      if(str[0] == "-"){
         r = false;
      }
      str = Math.abs(parseInt(str))
      
      if(str>Math.pow(2,31)-1) return r?Math.pow(2,31)-1:-Math.pow(2,31);
      return r?str:-str;
  };
