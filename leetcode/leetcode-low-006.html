兩個數組的交集 II

給定兩個數組，編寫一個函數來計算它們的交集。

輸入: nums1 = [1,2,2,1], nums2 = [2,2]
輸出: [2,2]

      
  /**
   * @param {number[]} nums1
   * @param {number[]} nums2
   * @return {number[]}
   */
  let intersect = function(nums1, nums2) {
      let res = [];
      let map = {};
      for (let e of nums1) {
          map[e] = map[e] + 1 || 1;
      }
      for (let e of nums2) {
          if (map[e]) {
              res.push(e);
              map[e]--;
          }
      }
      return res;
  };
