合並兩個有序數組

給定兩個有序整數數組 nums1 和 nums2，將 nums2 合並到 nums1 中，使得 num1 成為一個有序數組。

輸入:
nums1 = [1,2,3,0,0,0], m = 3
nums2 = [2,5,6],       n = 3

輸出: [1,2,2,3,5,6]
      
  /**
   * @param {number[]} nums1
   * @param {number} m
   * @param {number[]} nums2
   * @param {number} n
   * @return {void} Do not return anything, modify nums1 in-place instead.
   */
  var merge = function (nums1, m, nums2, n) {
      let i = m - 1,
          j = n - 1,
          k = m + n -1;
      
      while( j >= 0 && i >= 0){
          if(nums1[i] > nums2[j]){
              nums1[k--] = nums1[i--];
          }else{
              nums1[k--] = nums2[j--];
          }
      }
      while(j >= 0){
          nums1[k--] = nums2[j--];
      }
  };
