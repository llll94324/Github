整數反轉

給出一個 32 位的有符號整數，你需要將這個整數中每位上的數字進行反轉。

輸入: 123
輸出: 321

      
  /**
   * @param {number} x
   * @return {number}
   */
  var reverse = function(x) {
      var res = 0;
      var symbol = x > 0;
      x = x > 0?x:-x
      while(x != 0){
          
          res = res * 10 + x % 10;
          x = Math.floor(x / 10)
      }
      if(res>Math.pow(2,31)-1) return 0;
      return symbol?res:-res;
  };
